# Toggle

The toggle can be used when you need to switch a single item on or off.

<Canvas>
  <Story
    name="Default"
    args={{
      name: "default",
      isChecked: false,
      label: "Label",
      isDisabled: false,
    }}
    parameters={{
      controls: {
        exclude: ["name"],
      },
      docs: {
        source: {
          code: `
<div class="made-c-form__element">
  <span class="made-c-form__label" id="default">
    Label
  </span>
  <label class="made-c-toggle">
    <input type="checkbox" class="made-c-toggle__input" aria-labelledby="default" />
    <span class="made-c-toggle__slider"></span>
  </label>
</div>`,
        },
      },
    }}
  >
    >{toggle.bind({})}
  </Story>
</Canvas>
